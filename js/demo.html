Integration

Add a single script tag and use data-worldcredit attributes:

<!-- Inline badge next to a username -->
<span>Ryan N.</span>
<div data-worldcredit="ryannapp" data-style="inline"></div>

<!-- Pill badge -->
<div data-worldcredit="ryannapp" data-style="pill" data-theme="light"></div>

<!-- Card badge for sidebar -->
<div data-worldcredit="ryannapp" data-style="card"></div>

<!-- Shield (minimal) -->
<span>Ryan N.</span>
<div data-worldcredit="ryannapp" data-style="shield"></div>

<!-- Load SDK (place before </body>) -->
<script src="https://world-credit.com/sdk/badge.js"></script>
